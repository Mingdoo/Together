create database together;

use together;

create table user (
 id bigint auto_increment,
 user_id varchar(255),
 disability varchar(255),
 name varchar(255),
 nickname varchar(255),
 password varchar(255),
 email varchar(255),
 primary key(id)
 );
 
  create table conference_category (
  id bigint auto_increment,
  name varchar(255),
  primary key(id)
 );
 
create table conference (
 id bigint auto_increment,
 owner_id bigint,
 conference_category bigint,
 call_start_time DateTime,
 call_end_time DateTime,
 thumbnail_url varchar(255),
 title varchar(255),
 description varchar(255),
 is_active Boolean,
 primary key(id),
 foreign key(owner_id) references user (id),
 foreign key(conference_category) references conference_category (id)
 );
 
 create table conference_history (
  id bigint auto_increment,
  conference_id bigint,
  user_id bigint,
  action int,
  inserted_time Datetime,
  primary key(id),
  foreign key(conference_id) references conference (id),
  foreign key(user_id) references user (id)
 );
 
 create table user_conference (
  id bigint auto_increment,
  conference_id bigint,
  user_id bigint,
  primary key(id),
  foreign key(conference_id) references conference (id),
  foreign key(user_id) references user (id)
 );
 
 desc user;
 desc conference;
 desc user_conference;
 desc conference_history;
 desc conference_category;
